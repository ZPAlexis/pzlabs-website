public void GeneratePowerUpOptions() {
  List list = new List();
  for(int i = 0; i < powerUpList.Count; i++)
  {
      list.Add(i);
  }
  
  int rand1 = Random.Range(0, list.Count);
  int value1 = list[rand1];
  list.RemoveAt(rand1);
  InstantiatePowerUp(value1, choice1, cardPrefab, gameObject.transform);

  int rand2 = Random.Range(0, list.Count);
  int value2 = list[rand2];
  list.RemoveAt(rand2);
  InstantiatePowerUp(value2, choice2, cardPrefab, gameObject.transform);

  int rand3 = Random.Range(0, list.Count);
  int value3 = list[rand3];
  list.RemoveAt(rand3);
  InstantiatePowerUp(value3, choice3, cardPrefab, gameObject.transform);
}
